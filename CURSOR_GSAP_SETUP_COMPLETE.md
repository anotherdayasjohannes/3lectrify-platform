```tsx
// packages/animations/src/hooks/useParallax.ts
import { useRef } from 'react';
import { useGSAP } from '@gsap/react';
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { shouldReduceMotion } from '../utils/accessibility';

gsap.registerPlugin(ScrollTrigger);

interface ParallaxOptions {
  speed?: number;
  direction?: 'up' | 'down';
}

/**
 * Creates a parallax scroll effect on an element.
 * Respects prefers-reduced-motion for accessibility.
 */
export function useParallax<T extends HTMLElement = HTMLDivElement>(
  options: ParallaxOptions = {}
) {
  const { speed = 0.5, direction = 'up' } = options;
  const ref = useRef<T>(null);

  useGSAP(() => {
    if (!ref.current || shouldReduceMotion()) return;

    const yValue = direction === 'up' ? -100 : 100;

    gsap.to(ref.current, {
      y: `${yValue * speed}%`,
      ease: 'none',
      scrollTrigger: {
        trigger: ref.current,
        start: 'top bottom',
        end: 'bottom top',
        scrub: true,
        markers: process.env.NODE_ENV === 'development'
      }
    });
  }, { scope: ref });

  return ref;
}

// Don't forget to export from index.ts!
```

## üèÜ Animation Principles

### Our Standards
1. **Subtle > Flashy**: Enhance, don't distract
2. **Fast > Slow**: 0.3-0.8s sweet spot
3. **Purposeful**: Every animation tells the story
4. **Consistent**: Reuse easings and durations
5. **Accessible**: Never exclude users

### GSAP Best Practices
‚úÖ Always use `useGSAP` hook  
‚úÖ Add `once: true` to entrance animations  
‚úÖ Set explicit default visibility  
‚úÖ Respect `prefers-reduced-motion`  
‚úÖ Use `{ scope: containerRef }` for cleanup  
‚úÖ Add dev markers: `markers: process.env.NODE_ENV === 'development'`  
‚úÖ Test on mobile (reduce complexity)  

‚ùå Never use `useEffect` for GSAP  
‚ùå Never forget cleanup  
‚ùå Never animate width/height (use scale)  
‚ùå Never use permanent `will-change`  
‚ùå Never ignore accessibility
